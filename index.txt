function calculate(a, b) {
  const subtract = a - b;
  const multiply = a * b;
  const divide = a / b;

  return {subtract, multiply, divide}
}

{subtract, multiply, divide} = calculate(4, 7);

document.write("<p>Difference " + subtract + "</p>");
document.write("<p>Product: " + multiply + "</p>");
document.write("<p>Quotient " + divide + "</p>");
